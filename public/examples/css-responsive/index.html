纯 CSS

核心代码

.card { flex: 1 1 100%; }                    /* 手机: 1列 */

@media (min-width: 768px) {
  .card { flex: 1 1 calc(50% - 15px); }     /* 平板: 2列 */
}

@media (min-width: 992px) {
  .card { flex: 1 1 calc(25% - 15px); }     /* 桌面: 4列 */
}
